### for-loop

```python
#[0,15)
for i in range(15):
    print(i)

#[2,15)
for i in range(2,15):
    print(i)
```

### if-loop

```python
'''
if 判断条件1：
   执行语句1
elif 判断条件2：
   执行语句2
else 执行语句3
'''
a = int(input("请输入一个0<x<10的数："))
if a>5 and a<10:
    print("wow")
elif a > 2:
    print("zezeze")
else:
    print("yi~")
```

import random
x = int(input("剪刀：0，石头：1，布：2  ："))
y = random.randint(0,2)

if x == 0:
    if y == 0:
        print("平手")
    elif y == 1:
        print("you lose")
    else :
        print("you win")
elif x == 1:
    if y == 0:
        print("you win")
    elif y == 1:
        print("平手")
    else :
        print("you lose")
else :
    if y == 0:
        print("you lose")
    elif y == 1:
        print("you win")
    else :
        print("平手")
